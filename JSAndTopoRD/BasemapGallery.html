ArcGIS JS and Topo RD Basemap Gallery

Code: 
//create basemap gallery with a group id
var basemapGallery = new BasemapGallery({
    view: view,
    source: { 
        //GroupID Esri NL RD Basemaps
        query: "id:0a23b7ca2f0543bd89f29f6c28be4b4c" 
    } 
});

//create an expand with the basemap Gallery
var basemapExpand = new Expand({
                                view:view,
                                content:basemapGallery,
                                expanded:true
                            });

// Add the widget to the top-right corner of the view
view.ui.add(basemapExpand, {
    position: "top-right"
});
